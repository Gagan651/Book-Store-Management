CREATE TABLE BOOK_AUTHOR(
AUTHOR_ID NUMBER(20),
AUTHOR_NAME VARCHAR2(30) NOT NULL,
AUTHOR_MAIL_ID VARCHAR2(50) NOT NULL,
PRIMARY KEY (AUTHOR_ID)
 );




CREATE TABLE BOOKS(
BOOK_ID NUMBER(20) PRIMARY KEY,
PRICE NUMBER(20),
PUB_YEAR VARCHAR2(20) NOT NULL,
BOOK_NAME VARCHAR2 (50) NOT NULL,
AUTHOR_ID NUMBER(20),
QUANTITY NUMBER(20),
FOREIGN KEY(AUTHOR_ID) REFERENCES BOOK_AUTHOR(AUTHOR_ID)ON DELETE CASCADE);





CREATE TABLE CUSTOMER(
CUSTOMER_ID NUMBER(20) PRIMARY KEY,
CUSTOMER_NAME VARCHAR2(50) NOT NULL,
PHONE_NO NUMBER (10) NOT NULL,
ADDRESS VARCHAR2 (50) NOT NULL,
MAIL_ID VARCHAR2 (50));



CREATE TABLE PURCHASE(
CUSTOMER_ID NUMBER(20),
BOOK_ID NUMBER(20),
QUANTITY NUMBER(20),
TOTALPRICE NUMBER(20),
FOREIGN KEY(BOOK_ID) REFERENCES BOOKS(BOOK_ID)ON DELETE CASCADE,
FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID)ON DELETE CASCADE);



CREATE TABLE PAYMENT(
CUSTOMER_ID NUMBER(20),
PAYMENT_ID NUMBER(20) PRIMARY KEY,
MODE_OF_PAYMENT VARCHAR2(50) NOT NULL,
TOTALPRICE NUMBER(20),
FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID)ON DELETE CASCADE);


CREATE TABLE RATING(
CUSTOMER_ID NUMBER(20)PRIMARY KEY,
IN_STORE_EXPERIENCE VARCHAR2(10),
CUSTOMER_SUGGESTION VARCHAR2(200),
FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID)ON DELETE CASCADE);
